template <int S> SecureMod
pks(vector<SecureInt<S>> v, SecureInt<S> k)
{
    vector<SecureMod> r;
    for (size_t i=0; i < v.size(); i++)
    {
        auto eq = i == k;
        auto sel = eq * SecureMod(v[i]);
        r.push_back(sel);
    }
    return sum(r);
}
