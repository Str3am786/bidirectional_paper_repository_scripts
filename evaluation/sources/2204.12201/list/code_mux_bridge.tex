template <int S> SecureMod
mux(SecureInt<S> input, SecureInt<S> item,
    SecureInt<S> t, SecureInt<S> f)
{
    auto cond = input == item;
    return cond * SecureMod(t)
        + !cond * SecureMod(f);
}
